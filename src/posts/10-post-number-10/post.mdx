import React from 'react'

const LikeButton = () => {
  const [value, setValue] = React.useState(0)
  return (
    <Wrapper>

          <AiOutlineLike />
        </button>
        <p>
          Liked the post?
          <br />
          Click thumbs up few times
        </p>
      </div>
      <span>+{value}</span>
    </Wrapper>
  )
}

export default LikeButton
